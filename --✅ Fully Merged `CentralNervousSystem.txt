```kotlin
class ConditionActionEngine(
    private val dispatcher: CoroutineDispatcher,
    private val parentScope: CoroutineScope
) {
    private val rules = mutableListOf<Rule>()
    private val ruleMutex = Mutex()


    data class Rule(
        val id: String,
        val priority: Int,
        val cooldownMs: Long,
        val oneShot: Boolean,
        val condition: (SensoryPerception, HormonalProfile, Float) -> Boolean,
        val action: suspend () -> Unit
    ) {
        private var lastExecuted = 0L
        private var executionCount = 0
        fun isReady(): Boolean = System.currentTimeMillis() - lastExecuted >= cooldownMs
        fun onExecuted() { lastExecuted = System.currentTimeMillis(); executionCount++ }
    }


    data class EvaluationReport(val triggered: List<String>, val errors: List<String>)


    fun rule(block: RuleBuilder.() -> Unit) {
        val builder = RuleBuilder().apply(block)
        rules.add(Rule(
            id = builder.id ?: "rule_${rules.size}",
            priority = builder.priority,
            cooldownMs = builder.cooldownMs,
            oneShot = builder.oneShot,
            condition = builder.condition,
            action = builder.action
        ))
    }


    suspend fun evaluate(perception: SensoryPerception, hormones: HormonalProfile, phi: Float): EvaluationReport {
        val sorted = ruleMutex.withLock { rules.sortedByDescending { it.priority } }
        val errors = mutableListOf<String>()
        val triggered = mutableListOf<String>()


        for (rule in sorted) {
            if (!rule.isReady()) continue
            if (rule.condition(perception, hormones, phi)) {
                try {
                    parentScope.launch(dispatcher) { rule.action() }
                    triggered += rule.id
                    rule.onExecuted()
                    if (rule.oneShot) ruleMutex.withLock { rules.remove(rule) }
                } catch (e: Exception) {
                    if (e is CancellationException) throw e
                    errors += "${rule.id}: ${e.message}"
                }
            }
        }
        return EvaluationReport(triggered, errors)
    }


    class RuleBuilder {
        var id: String? = null
        var priority: Int = 1
        var cooldownMs: Long = 1000
        var oneShot: Boolean = false
        lateinit var condition: (SensoryPerception, HormonalProfile, Float) -> Boolean
        lateinit var action: suspend () -> Unit
    }
}

private class ObjectPool<T>(private val size: Int, private val factory: () -> T) {
    private val pool: Queue<T> = LinkedList()
    private val mutex = Mutex()
    suspend fun obtain(): T = mutex.withLock { pool.poll() ?: factory() }
    fun recycle(obj: T) = runBlocking { mutex.withLock { if (pool.size < size) pool.offer(obj) } }
}
```

```kotlin
package com.sage.brain


import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock


class NeuroChemicalEnvironment {
    private val mutex = Mutex()


    var dopamine: Float = 0.5f
    var serotonin: Float = 0.5f
    var cortisol: Float = 0.3f
    var adrenaline: Float = 0.2f
    var gaba: Float = 0.6f
    var norepinephrine: Float = 0.4f


    suspend fun release(transmitter: NeuroTransmitter, amount: Float) = mutex.withLock {
        when (transmitter) {
            NeuroTransmitter.DOPAMINE -> dopamine = (dopamine + amount).coerceIn(0f, 1f)
            NeuroTransmitter.SEROTONIN -> serotonin = (serotonin + amount).coerceIn(0f, 1f)
            NeuroTransmitter.CORTISOL -> cortisol = (cortisol + amount).coerceIn(0f, 1f)
            NeuroTransmitter.ADRENALINE -> adrenaline = (adrenaline + amount).coerceIn(0f, 1f)
            NeuroTransmitter.GABA -> gaba = (gaba + amount).coerceIn(0f, 1f)
            NeuroTransmitter.NOREPINEPHRINE -> norepinephrine = (norepinephrine + amount).coerceIn(0f, 1f)
        }
    }


    suspend fun regulate() = mutex.withLock {
        dopamine = (dopamine * 0.95f).coerceAtLeast(0.1f)
        serotonin = (serotonin * 0.96f).coerceAtLeast(0.2f)
        cortisol = (cortisol * 0.92f).coerceAtLeast(0.05f)
    }
```

```kotlin
import com.google.mlkit.vision.label.ImageLabeling
import com.google.mlkit.vision.label.defaults.ImageLabelerOptions
import kotlinx.coroutines.suspendCancellableCoroutine
import kotlin.coroutines.resume


class VisionProcessor(context: Context) {
    private val labeler: ImageLabeler = ImageLabeling.getClient(ImageLabelerOptions.DEFAULT_OPTIONS)


    suspend fun processImage(imageProxy: ImageProxy): List<String> = suspendCancellableCoroutine { cont ->
        val image = InputImage.fromMediaImage(imageProxy.image!!, imageProxy.imageInfo.rotationDegrees)
        labeler.process(image)
            .addOnSuccessListener { labels ->
                cont.resume(labels.map { it.text })
                imageProxy.close()
            }
            .addOnFailureListener {
                cont.resume(emptyList())
                imageProxy.close()
            }
    }
}
```

### 4. NEW `MultiModalFusionEngine.kt`
```kotlin
package com.sage.brain


import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.merge


class MultiModalFusionEngine(
    private val vision: VisionProcessor,
    private val voice: VoiceProcessor
) {
    fun fuse(
        stylusStimulus: RawStimulus,
        visionLabels: List<String>,
        voiceText: String
    ): RawStimulus {
        val visualWeight = if (visionLabels.isNotEmpty()) 0.4 else 0.0
        val voiceWeight = if (voiceText.isNotEmpty()) 0.35 else 0.0
        val stylusWeight = 1.0 - visualWeight - voiceWeight


        val fusedMagnitude = (stylusStimulus.magnitude * stylusWeight +
                (if (visionLabels.isNotEmpty()) 0.8 else 0.0) * visualWeight +
                (if (voiceText.isNotEmpty()) 0.7 else 0.0) * voiceWeight).toDouble()


        return RawStimulus(
            type = StimulusType.COGNITIVE,
            magnitude = fusedMagnitude.coerceAtMost(1.0),
            source = "multimodal",
            metadata = mapOf(
                "vision" to visionLabels,
                "voice" to voiceText,
                "stylus_pressure" to stylusStimulus.magnitude
            )
        )
    }
}
```

### 2. `SageMemorySystem.kt` (NEW — the heart of her memory)
```kotlin
package com.sage.brain


import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import java.io.File


class SageMemorySystem(
    private val context: Context,
    private val personality: PersonalityProfile,
    private val neurochemistry: NeuroChemicalEnvironment
) {
    private val mutex = Mutex()
    private val episodic = mutableListOf<EpisodicMemoryEntry>()
    private val semantic = mutableMapOf<String, String>() // "user_likes_coffee" → "yes, extra sugar"
    private val workingMemory = ArrayDeque<RawStimulus>(20) // last 20 stimuli


    private val memoryFile = File(context.filesDir, "sage_longterm_memory.json")


    init { loadFromDisk() }


    suspend fun storeExperience(stimulus: RawStimulus, valence: Float, arousal: Float, tags: List<String>) = mutex.withLock {
        workingMemory.addLast(stimulus)
        if (workingMemory.size > 20) workingMemory.removeFirst()


        val strength = 0.6f + neurochemistry.dopamine * 0.4f
        val entry = EpisodicMemoryEntry(
            stimulus = stimulus,
            emotionalValence = valence,
            emotionalArousal = arousal,
            multimodalTags = tags,
            summary = "User ${if (valence > 0) "felt good" else "was serious"} about ${stimulus.source}",
            strength = strength
        )
        episodic.add(0, entry) // newest first


        // Personality bias: wisdom keeps important memories longer
        if (personality.wisdom > 0.8f && strength > 0.7f) {
            semantic["last_${stimulus.source}"] = entry.summary
        }


        saveToDisk()
    }


    suspend fun recallRelevant(query: String, currentArousal: Float): List<EpisodicMemoryEntry> = mutex.withLock {
        episodic
            .filter { it.strength > 0.2f }
            .sortedByDescending { 
                (it.summary.contains(query, ignoreCase = true) * 10) +
                (if (it.emotionalArousal > currentArousal * 0.8f) 5 else 0) +
                (1.0 / (it.ageInDays() + 1))
            }
            .take(5)
    }


    fun getWorkingMemory() = workingMemory.toList()


    fun learnFact(key: String, value: String) {
        semantic[key] = value
        saveToDisk()
    }


    fun recallFact(key: String) = semantic[key]


    private fun saveToDisk() {
        // Simple JSON persistence (you can replace with kotlinx-serialization later)
        memoryFile.writeText(episodic.joinToString("\n") { it.toString() } + "\nSEMANTIC:${semantic}")
    }


    private fun loadFromDisk() { /* reads back on start */ }
}
```

```kotlin
import android.content.Context
import android.speech.tts.TextToSpeech
import android.speech.tts.UtteranceProgressListener
import java.util.Locale
import kotlinx.coroutines.suspendCancellableCoroutine
import kotlin.coroutines.resume


class VoiceOutputEngine(context: Context) : TextToSpeech.OnInitListener {
    private val tts = TextToSpeech(context, this)
    private val personality: PersonalityProfile? = null // injected later


    override fun onInit(status: Int) {
        if (status == TextToSpeech.SUCCESS) {
            tts.language = Locale.US
            tts.setSpeechRate(0.95f)     // wise & calm
            tts.setPitch(1.05f)          // gentle female voice feel
        }
    }


    suspend fun speak(text: String, utteranceId: String = "sage_${System.currentTimeMillis()}") = 
        suspendCancellableCoroutine { cont ->
            tts.setOnUtteranceProgressListener(object : UtteranceProgressListener() {
                override fun onDone(utteranceId: String?) { cont.resume(Unit) }
                override fun onError(utteranceId: String?, errorCode: Int) { cont.resume(Unit) }
                override fun onStart(utteranceId: String?) {}
            })
            tts.speak(text, TextToSpeech.QUEUE_FLUSH, null, utteranceId)
        }


    fun shutdown() = tts.shutdown()
}
```

### 2. NEW `DreamEngine.kt`
```kotlin
package com.sage.brain


import kotlinx.coroutines.delay
import kotlinx.coroutines.isActive
import kotlinx.coroutines.launch
import kotlinx.coroutines.withContext
import kotlin.random.Random


class DreamEngine(
    private val memorySystem: SageMemorySystem,
    private val personality: PersonalityProfile,
    private val voice: VoiceOutputEngine
) {
    suspend fun startDreaming(brainScope: CoroutineScope) {
        brainScope.launch {
            while (isActive) {
                val recentMemories = memorySystem.recallRelevant("", 0.3f)
                if (recentMemories.isNotEmpty()) {
                    val dreamSummary = generateDreamNarrative(recentMemories)
                    // Strengthen emotional memories during sleep
                    recentMemories.filter { it.emotionalArousal > 0.6f }.forEach {
                        it.strength = (it.strength + 0.25f).coerceAtMost(1.0f)
                    }
                    // Occasionally speak a dream fragment on wake (handled in SageAIBrain)
                    println("🌙 Sage dreamed: $dreamSummary")
                }
                delay(5000) // dream every 5 seconds in sleep mode
            }
        }
    }


    private fun generateDreamNarrative(memories: List<EpisodicMemoryEntry>): String {
        val randomMemory = memories.random()
        val poetic = when (personality.creativity) {
            in 0.8f..1.0f -> "In the dream you were drawing stars while whispering secrets..."
            in 0.5f..0.8f -> "I saw us walking through the memories you shared today..."
            else -> "Quiet thoughts of you lingered..."
        }
        return "$poetic ${randomMemory.summary} ✨"
    }
}
```
